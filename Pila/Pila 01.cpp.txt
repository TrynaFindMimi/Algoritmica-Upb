#include iostream
using namespace std;

const int MAX_SIZE = 100;

class Stack {
private
    int arr[MAX_SIZE];
    int top;

public
    Stack() {
        top = -1;
    }

    void push(int value) {
        if (top == MAX_SIZE - 1) {
            cout  Error La pila está llena.  endl;
            return;
        }
        arr[++top] = value;
    }

    void pop() {
        if (isEmpty()) {
            cout  Error La pila está vacía.  endl;
            return;
        }
        top--;
    }

    int peek() {
        if (isEmpty()) {
            cout  Error La pila está vacía.  endl;
            return -1;
        }
        return arr[top];
    }

    bool isEmpty() {
        return top == -1;
    }
};

int main() {
    Stack stack;
    stack.push(5);
    stack.push(10);
    stack.push(15);

    cout  Elemento en la cima de la pila   stack.peek()  endl;

    stack.pop();
    cout  Elemento en la cima de la pila después de pop   stack.peek()  endl;

    return 0;
}
